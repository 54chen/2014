    riak_core:join(Node) => %%еҺ»еҲ°riak_core
    riak_core_gossip:legacy_gossip() => %%йҖҡзҹҘжЈҖжҹҘжҳҜеҗҰжҳҜ legacy gossipпјҢ
    riak_core:standard_join(Node,Ring,false) => %%йқһrejoin йқһlegacy
    riak_core_gossip:send_ring(Node, node()) => %% иҝӣе…ҘgossipжЁЎеқ—еҸ‘иө·ringиҜ·жұӮ
    gen_server:cast({?MODULE, FromNode}, {send_ring_to, ToNode}) => %%е№ҝиҖҢе‘Ҡд№Ӣ
    riak_core_gossip:handle_cast({send_ring_to, Node}, State) => %%ж”¶еҲ°
    gen_server:cast({?MODULE, Node}, {reconcile_ring, RingOut}) => %%е‘ҠиҜүе®ғд»¬ејҖе§ӢеҚҸи°ғ
    riak_core_gossip:handle_cast({reconcile_ring, RingIn}, State) => %%ж”¶еҲ°
    riak_core_ring_manager:ring_trans(fun reconcile/2, [OtherRing]) => %%з”ұдёҖдёӘеҚҸи°ғеҮҪж•°дј е…ҘеҲ°ring managerдёӯеҺ»жү§иЎҢ
    riak_core_ring_manager:prune_write_notify_ring => %%жҗһе®ҡдёҖдёӘж–°зҡ„ringж–Үд»¶пјҢйҖҡзҹҘжүҖжңүзҡ„listener
    riak_core_ring_events:ring_update(Ring) => %%йҖҡиҝҮeventйҖҡзҹҘ
    riak_core_ring_handler:handle_event({ring_update, Ring}, State) => %%жҺҘеҲ°event
    riak_core_vnode_manager:ring_changed(Ring) => %%vnode_managerйҖҡзҹҘеӨ§е®¶зҺҜдҝ®ж”№дәҶ

    riak_core_vnode:trigger_handoff(Pid, TargetNode) => %%з”ұvnodeжқҘе№Іжҙ»

    riak_core_handoff_manager:add_outbound => %%еҸ‘handoff
    handle_call({add_outbound,Mod,Idx,Node,Pid},_From,State=#state{handoffs=HS}) => %%ж”¶еҲ°иҜ·жұӮ
    riak_core_handoff_manager:send_handoff => %%еҸ‘
    riak_core_handoff_sender_sup:start_sender => %%еҗҜеҠЁsender жңҖз»ҲеҗҜеҠЁдәҶ riak_core_vnode_master:sync_commandзӣҙеҲ°еҗҢжӯҘз»“жқҹгҖӮ

    riak_core_vnode:mark_handoff_complete жӯӨд»Јз Ғз”ҹеӯҳеңЁriak_core_vnodeпјҢиҖҢдёҚжҳҜriak_core_vnode_managerпјҢеӣ дёәring_transжҳҜдёҖдёӘеҗҢжӯҘзҡ„callеҲ°ring managerзҡ„иҝҮзЁӢпјҢblockдёҖдёӘеҚ•зӢ¬зҡ„vnodeиҰҒжҜ”blockж•ҙдёӘvnode managerиҰҒеҘҪгҖӮblockиҝҷдёӘmanagerдјҡеҪұе“ҚжүҖжңүзҡ„vnodeгҖӮжӯӨд»Јз ҒеҜ№еӨҡдёӘе№¶иЎҢзҡ„vnodeжү§иЎҢжҳҜе®үе…Ё
    зҡ„,еӣ дёәйқ еҚ•дёӘ ring managerжҸҗдҫӣзҡ„еҗҢжӯҘеҢ–дҝқиҜҒдәҶжӢҘжңүжүҖжңүзҡ„зҺҜзҡ„еҸҳжӣҙз»ҸиҝҮ гҖӮ

    riak_core_vnode:vnode_command жҙ»еҠЁзҡ„vnodeиҝҗиЎҢдәҺдёүз§ҚзҠ¶жҖҒдёӯпјҡжӯЈеёёгҖҒhandoffгҖҒжҺЁиҝӣгҖӮ
    еңЁжӯЈеёёзҠ¶жҖҒпјҢvnodeе‘Ҫд»Өйқ handle_commandжқҘдј йҖ’гҖӮеҪ“дёҖдёӘhandoffиў«и§ҰеҸ‘пјҢhandoff_nodeиў«и®ҫзҪ®дёәзӣ®ж ҮиҠӮзӮ№пјҢе№¶дё”иҝҷдёӘvnodeиў«з§°еҒҡиҝӣе…ҘдәҶhandoffзҠ¶жҖҒгҖӮ
    еңЁhandoffзҠ¶жҖҒпјҢvnodeе‘Ҫд»Өдҫқйқ handle_handoff_commandжқҘдј йҖ’гҖӮеҚідҪҝhandoffзЁӢеәҸжҳҜйқһblockзҡ„пјҲжҜ”еҰӮеңЁriak_kvдёӯжІЎжңүдҪҝз”Ёasync foldж—¶пјүпјҢеңЁhandoffжңҹй—ҙпјҢдёҖдёӘvnodeд№ҹдјҡиў«blockдҪҸпјҲжүҖд»ҘжІЎжңүжңҚеҠЎе‘Ҫд»ӨпјүгҖӮ
    handoffзҠ¶жҖҒд№ӢеҗҺпјҢдёҖдёӘvnodeдјҡиҝӣе…ҘеҲ°жҺЁиҝӣзҠ¶жҖҒгҖӮжҺЁиҝӣзҠ¶жҖҒжҳҜж–°зҡ„gossip/membershipд»Јз Ғзҡ„дә§зү©пјҢе№¶дё”legacyжЁЎејҸзҡ„иҠӮзӮ№йҮҢдёҚдјҡеҮәзҺ°гҖӮжҺЁиҝӣзҠ¶жҖҒд»ЈиЎЁдәҶvnodeе·Із»Ҹдј йҖ’иҮӘе·ұж•°жҚ®еҲ°ж–°иҠӮзӮ№зҡ„жғ…еҶөпјҢдҪҶж–°зҡ„иҠӮзӮ№е№¶жңӘеңЁзҺҜдёҠеҲ—еҮәеҪ“еүҚиҠӮзӮ№зҡ„дҝЎжҒҜгҖӮиҝҷдјҡеҮәзҺ°жҳҜеӣ дёәеўһеҠ зҡ„vnodeд»Қз„¶еңЁзӯүhandoffд»–д»¬зҡ„ж•°жҚ®з»ҷж–°зҡ„жӢҘжңүиҖ…пјҢжҲ–иҖ…еҸӘжҳҜеӣ дёәзҺҜиҝҳжұҮиҒҡеңЁиҝҷдёӘж–°иҠӮзӮ№д№ӢдёҠгҖӮеңЁжҺЁиҝӣзҠ¶жҖҒпјҢжүҖжңүзҡ„vnodeжҢҮд»Өе’Ңcoverageе‘Ҫд»ӨйғҪдјҡжҢүиҝҮзЁӢжҺЁз»ҷж–°зҡ„жӢҘжңүиҖ…гҖӮ

    riak_core_vnode:start_manager_event_timer зӢ¬жңүзҡ„vnodeиҝӣзЁӢдёҺvnode managerз»“еҗҲзҙ§еҜҶгҖӮеҪ“vnodeзҡ„дәӢд»¶и§ҰеҸ‘пјҢиҝҷдёӘvnodeеҝ…йЎ»зЎ®дҝқиҝҷдёӘдәӢд»¶иў«еҸ‘йҖҒеҲ°vnode managerпјҢиҝҷе°Ҷдјҡдә§з”ҹдёҖдёӘзҠ¶жҖҒеҸҳеҢ–зҡ„еҶіе®ҡпјҢе№¶дё”еҸ‘еӣһйҖӮеҪ“зҡ„дҝЎжҒҜз»ҷиҝҷдёӘvnodeгҖӮдёәдәҶжңҖеӨ§йҷҗеәҰең°еҮҸе°‘йҳ»еЎһпјҢдҪҝз”ЁдәҶејӮжӯҘзҡ„ж¶ҲжҒҜгҖӮvnode managerжҢӮжҺүиҖҢдёўеӨұйқ vnodeеҸ‘йҖҒзҡ„дҝЎжҒҜжҳҜеҸҜиғҪзҡ„гҖӮеӣ жӯӨпјҢvnodeе‘ЁжңҹжҖ§ең°йҮҚеҸ‘дәӢд»¶ж¶ҲжҒҜпјҢзӣҙеҲ°дёҖдёӘд»Һvnode managerжқҘзҡ„еҗҲйҖӮзҡ„ж¶ҲжҒҜиў«жҺҘж”¶гҖӮ
